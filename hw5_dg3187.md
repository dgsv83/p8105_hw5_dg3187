``` r
t_esti_pvalue = function(n = 30, mu, sigma = 5) { #default sample size and sd
  
  sim_data =
    tibble(
     x = rnorm(n, mean = mu, sd = sigma)
    )
  
  sim_data %>%
    summarize(
      t_test = t.test(x, mu = mu, conf.level = 0.95) %>% 
        broom::tidy() %>% 
        select(p.value, estimate)
  )
  
}
```
